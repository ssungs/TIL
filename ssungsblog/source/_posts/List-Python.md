# 리스트

리스트는 대괄호( [ ] )로 감싸고 요솟값은 쉼표( , )로 구분

비어 있는 리스트는 변수 = list() 로 생성가능

리스트 안에 요솟값들은 순서를 가지면서 변경이 가능

```
a = [1, 2, 3, 4, 5, 6, 7, 8, 9]
a1 = [1, 3, 5, 7, 9]
b = ['a', 'b', 'c']
c = [1, 'a', 2, 'b']          #숫자와 문자는 같은 리스트 안에 포함가능
d = [1, 2, ['a', 'b']]        #리스트 안에 리스트도 포함가능
e = [1, 2, ['a', 'b',['a', 'b', 'c']]]   #3중이상의 구조도 가능
```

## 인덱싱과 슬라이싱

### 리스트의 인덱싱(리스트내에 요솟값 찾기) >> 변수[찾을 리스트 번호]

```
a[0]
>>1
b[-1]
>>'c'
c[0] + c[2]
>>3
d[2]          #리스트 번호로 안에 있는 리스트 찾기
>>['a', 'b']
d[2][0]      #리스트 안에 리스트 내에 요솟값 찾기 (변수[찾을요솟값][찾을요솟값])
>>'a'
e[2][2][1]   #3중구조도 동일하게 찾기가능
>>'b'
```

### 리스트의 슬라이싱(리스트내에 요솟값 잘라내기) >> 변수[시작번호:종료번호]

```
a[0:3]
>>[1, 2, 3]
a[3:]           # 3 <= a 끝까지
>>[4, 5, 6, 7, 8, 9]
a[:5]           # 0<= a < 5 0부터 4까지 자르기
>>[1, 2, 3, 4, 5]
e[2][2][:2]     # 3중 이상구조에서는 리스트안의 리스트는 번호로 호출
>>['a', 'b']    # 찾을 리스트에서만 (:)가능
리스트 연산하기
a + a1    # 리스트끼리 더하면 뒤에 적은 변수의 리스트들이 뒤로 붙음
>>[1, 2, 3, 4, 5, 6, 7, 8, 9, 1, 3, 5, 7, 9]
a + b     # 리스트내에서는 숫자와 문자 더하기 가능 순서로 이루어졌기 때문
>>[1, 2, 3, 4, 5, 6, 7, 8, 9, 'a', 'b', 'c']
b * 3    # 리스트 곱하기는 문자열과 동일
>>['a', 'b', 'c', 'a', 'b', 'c', 'a', 'b', 'c']
len(a)    # len 함수는 리스트내의 요솟값 갯수를 나타냄
>>9
len(e)   # 리스트내의 리스트는 요솟값의 하나 이므로 갯수는 3이 됨
>>3
```

## 리스트의 수정과 삭제

```
A = [1, 2, 3]
A[2] = 7
A                # 요솟값을 다른값으로 변경하게 되면 그대로 리스트에 저장됨
>>[1, 2, 7]
del 함수를 사용해 리스트 요소 삭제하기
del A[2]
A              # 요솟값을 삭제하게 되면 그대로 리스트에 저장됨
>>[1, 2]
B = [1, 9, 6, 2, 4, 8, 5, 3, 7]
del B[3:]
B             # 요솟값의 범위를 지정해서 삭제 가능
>>[1, 9, 6]
```

## 리스트 관련 함수들

### 리스트에 요소 추가

```
A.append(3)          # append 함수는 리스트에 요솟값을 추가하되 리스트 마지막 순서에 추가
A
>>[1, 2, 3]
A.append([4, 5, 6]) 
A          # 리스트에 리스트를 추가하는 것도 가능 마찬가지로 마지막 순서로 추가
>>[1, 2, 3, [4, 5, 6]]
B.sort()
B                  # sort 함수는 내림차순 정렬
>>[1, 2, 3, 4, 5, 6, 7, 8, 9]
B.reverse()
B                 # reverse 함수는 올림차순 정렬
>>[9, 8, 7, 6, 5, 4, 3, 2, 1]
B.sort(reverse=True)
B                # sort 함수는 reverse=False 를 내장하고 있고 True로 바꿔주면
                 # reverse 와 동일한 함수로 사용 가능
>>[9, 8, 7, 6, 5, 4, 3, 2, 1]
A.index(3)      # index(요솟값) 을 넣으면 요솟값의 위치 번호가 출력
>>2
A.index([4, 5, 6])   # 리스트내의 리스트 위치 번호도 출력 가능
>>3
A.index([4, 5])  
>> # 리스트내의 요솟값을 전부 적지 않으면 리스트내에서 찾기 불가능
C  = [1, 2, 3]
C.insert(0, 9)
C                  # insert(넣을 위치, 넣을 요솟값) 위치 , 값 순서
>>[9, 1, 2, 3]
C.insert(2, 1)
C
>>[9, 1, 1, 1, 2, 3]
C.remove(1)
C           # remove(요솟값) 요솟값을 지움, 중복인 경우 가장 앞 순서 하나만 지움
>>[9, 1, 1, 2, 3]
C.pop()         # pop(빼낼 순서) ( ) 공백으로 둘 경우 가장 마지막 순서를 빼냄
>>3
C              # pop으로 빼내면 리스트에서 사라짐 
>>[9, 1, 1, 2]
D = C.pop(2)
D             # pop으로 빼낸 요소를 사용하는 경우 변수를 지정해서 사용
>>1
C
>>[9, 1, 2]
E = [1, 2, 3, 1, 5, 1]
E.count(1)   # count(요솟값) 을 하면 리스트내의 요솟값 갯수를 출력
>>3
C.extend([4,5,6])
C # extend([리스트]) 는 ( ) 안에 리스트만 넣을 수 있으며 제일 마지막 위치에 추가
>>[9, 1, 2, 4, 5, 6]
# extend 와 append 의 차이점은 extend 는 리스트에 요솟값을 포함 시키는 것이고
# append는 리스트내에 리스트를 포함시킬 수 있는 것은 동일하나 리스트내에 리스트가 포함됨
```
